    <!-- Scripts -->
    <script>
        // Menu Toggle
        const menuBtn = document.getElementById('menu-btn');
        const mobileMenu = document.getElementById('mobile-menu');
        const backToTop = document.getElementById('backToTop');

        menuBtn.addEventListener('click', () => {
            mobileMenu.classList.toggle('hidden');
        });

        // Close menu on link click
        document.querySelectorAll('.mobile-link').forEach(link => {
            link.addEventListener('click', () => mobileMenu.classList.add('hidden'));
        });

        // Back to Top functionality
        window.onscroll = function() {
            if (document.body.scrollTop > 300 || document.documentElement.scrollTop > 300) {
                backToTop.classList.remove('opacity-0', 'invisible', 'scale-50');
                backToTop.classList.add('opacity-100', 'visible', 'scale-100');
            } else {
                backToTop.classList.add('opacity-0', 'invisible', 'scale-50');
                backToTop.classList.remove('opacity-100', 'visible', 'scale-100');
            }
        };

        backToTop.addEventListener('click', () => {
            window.scrollTo({
                top: 0,
                behavior: 'smooth'
            });
        });
    </script>
